## 创建对象
#### 工厂模式
``` javascript
function Person(name, age){
    var o = new Object(){
        o.name = name;
        o.age = age;
        o.sayName = function(){
            console.log(this.name);
        }
    }
    return o;
}
var p = Person("Nicholas", 29);  // 注意没有new
```
#### 构造函数模式
``` javascript
function Person(name, age){
    this.name = name;
    this.age = age;
    this.sayName = function(){
        console.log(this.name);
    }
}
var p = new Person("Nicholas", 29);
```
#### 原型模式
``` javascript
function Person(){
}
Person.prototype = {
    name: "Nicholas",
    age: 29,
    sayName: function(){
        console.log(this.name);
    }
}
var p = new Person();
```
#### 组合使用构造函数模式和原型函数
最广泛、认同度最高的一种自定义类型的方法
``` javascript
// 定义实例属性
function Person(name, age){
    this.name = name;
    this.age = age;
}
// 定义方法和共享属性
Person.prototype = {
    constructor: Person,
    sayName: function(){
        console.log(this.name);
    }
}
var p = new Person("Nicholas", 29);
```
